- Song selection
[(@SongIndex:*)
{@SongText:#PART(@SongIndex,@Songs)#}
- update other stream deck profile
{@Profile:#@SongText#}
- manage rendering with line break
{@SongTextML:#SUBSTITUTE(@SongText," ",CHAR(10))#}
- compute previous and next text
{@SongTextPrev:#PART(@SongIndex-1,@Songs)#}
{@SongTextNext:#PART(@SongIndex+1,@Songs)#}
- manage out of limit and rendering with line break
{@SongPrevOOL:#IF(@SongTextPrev=@SongText,1,0)#}
{@SongTextMLPrev:#IF(@SongPrevOOL=1," ",SUBSTITUTE(@SongTextPrev," ",CHAR(10)))#}
{@SongNextOOL:#IF(@SongTextNext=@SongText,1,0)#}
{@SongTextMLNext:#IF(@SongNextOOL=1," ",SUBSTITUTE(@SongTextNext," ",CHAR(10)))#}
]
